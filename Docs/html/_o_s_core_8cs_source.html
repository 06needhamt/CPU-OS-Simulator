185  private SchedulerFlags? CreateSchedulerFlags()
186  {
187  SchedulerFlags temp = new SchedulerFlags();
188  temp.schedulingPolicies = schedulingPolicy;
189  if (temp.schedulingPolicies == EnumSchedulingPolicies.ROUND_ROBIN) // if the user selected round robin scheduling
190  {
191  temp.RR_Priority_Policy = priorityPolicy;
192  temp.RR_TimeSlice = RR_Time_Slice;
193  temp.RR_Type = roundRobinType;
194  temp.TimeSliceUnit = RR_Time_Slice_Unit;
195  }
196  temp.allowCPUAffinity = allowCPUAffinity;
197  temp.defaultScheduler = useDefaultScheduler;
198  temp.runningWithNoProcesses = runWithNoProcesses;
199  temp.usingSingleCPU = true; //TODO Change this when multiple CPU's are Implemented
200  if (!temp.runningWithNoProcesses) // if the user selected to run the scheduler no processes
201  {
202  temp.readyQueue = readyQueue;
203  temp.waitingQueue = waitingQueue;
204  SimulatorProcess proc = readyQueue.Dequeue();
205  if (proc != null)
206  {
207  proc.CurrentState = EnumProcessState.RUNNING;
208  }
209  temp.runningProcess = proc; // populate the queues and set the first processes state to running
210  }
211  else // otherwise create a blank queue and set the running process to null
212  {
213  readyQueue = new Queue<SimulatorProcess>();
214  waitingQueue = new Queue<SimulatorProcess>();
215  temp.runningProcess = null;
216  }
217  temp.cpuClockSpeed = CPUClockSpeed;
218  temp.issuedLotteryTickets = new List<LotteryTicket>();
219  temp.drawnLotteryTickets = new List<LotteryTicket>();
220  temp.core = this;
221  return temp;
222  }
