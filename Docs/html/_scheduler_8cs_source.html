183  private async void RunScheduler()
184  {
185  while (readyQueue.Count > 0 && readyQueue.Peek() != null && !suspended)
186  {
187  switch (schedulingPolicy)
188  {
189  case EnumSchedulingPolicies.FIRST_COME_FIRST_SERVED:
190  {
191  long life = CalculateLifetime();
192  ExecuteFirstComeFirstServed(life);
193  break;
194  }
195  case EnumSchedulingPolicies.SHORTEST_JOB_FIRST:
196  {
197  long life = CalculateLifetime();
198  ExecuteShortestJobFirst(life);
199  break;
200  }
201  case EnumSchedulingPolicies.ROUND_ROBIN:
202  {
203  switch (RR_Priority_Policy)
204  {
205  case EnumPriorityPolicy.NON_PRE_EMPTIVE:
206  {
207  readyQueue = new Queue<SimulatorProcess>(readyQueue.OrderBy(x => x.ProcessPriority));
208  long life = CalculateLifetime();
209  int timeout = CalculateTimeSlice();
210  await CallFromMainThread(UpdateInterface);
211  ExecuteRoundRobin(timeout,life,false);
212  break;
213  }
214  case EnumPriorityPolicy.NO_PRIORITY:
215  {
216  long life = CalculateLifetime();
217  int timeout = CalculateTimeSlice();
218  ExecuteRoundRobin(timeout,life,false);
219  break;
220  }
221  case EnumPriorityPolicy.PRE_EMPTIVE:
222  {
223  readyQueue = new Queue<SimulatorProcess>(readyQueue.OrderBy(x => x.ProcessPriority));
224  long life = CalculateLifetime();
225  int timeout = CalculateTimeSlice();
226  await CallFromMainThread(UpdateInterface);
227  ExecuteRoundRobin(timeout,life,true);
228  break;
229  }
230  case EnumPriorityPolicy.UNKNOWN:
231  {
232  MessageBox.Show("Unknown priority policy selected");
233  return;
234  }
235  default:
236  {
237  MessageBox.Show("An unknown error occurred"); // TODO give better error message
238  return;
239  }
240  }
241  break;
242  }
243  case EnumSchedulingPolicies.LOTTERY_SCHEDULING:
244  {
245  break;
246  }
247  case EnumSchedulingPolicies.FAIR_SHARE_SCEDULING:
248  {
249  break;
250  }
251  case EnumSchedulingPolicies.UNKNOWN:
