214  switch (schedulingPolicy)

224  case EnumSchedulingPolicies.SHORTEST_JOB_FIRST:

229  DeallocateProcessMemory(runningProcess);

234  switch (RR_Priority_Policy)
235  {
236  case EnumPriorityPolicy.NON_PRE_EMPTIVE:

238  readyQueue = new Queue<SimulatorProcess>(readyQueue.OrderBy(x => x.ProcessPriority));

243  ExecuteRoundRobin(timeout,life,false);
244  if(runningProcess.Unit.Done)
245  DeallocateProcessMemory(runningProcess);
246  break;
247  }
248  case EnumPriorityPolicy.NO_PRIORITY:
249  {
250  long life = CalculateLifetime();
251  int timeout = CalculateTimeSlice();
252  ExecuteRoundRobin(timeout,life,false);
253  if(runningProcess.Unit.Done)
254  DeallocateProcessMemory(runningProcess);
255  break;
256  }
