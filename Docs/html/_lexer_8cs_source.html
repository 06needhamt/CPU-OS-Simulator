368  string[] temp;

376  tokenStrings = new LinkedList<string>(temp);
377  currentTokenString = tokenStrings.First;
378  nextTokenString = currentTokenString.Next;
379  previousTokenString = currentTokenString.Previous;
380  while (currentTokenString != null)
381  {
382  Token token = new GenericToken(currentTokenString.Value);
383  if (token.isOperator())
384  {
385  Operator op = new Operator(currentTokenString.Value);
386  op.Type = op.DetectType();
387  tokens.AddLast(new LinkedListNode<Token>(op));
388  }
389  else if (token.isKeyword())

391  Keyword keyword = new Keyword(currentTokenString.Value);

393  tokens.AddLast(new LinkedListNode<Token>(keyword));
394  }
395  else if (token.isType())
396  {
397  Typename typename = new Typename(currentTokenString.Value);
398  typename.Type = typename.DetectType();
399  tokens.AddLast(new LinkedListNode<Token>(typename));
400  }

403  token.Type = token.DetectType();
404  if ((EnumTokenType) token.Type == EnumTokenType.STRING)

406  StringLiteral literal = new StringLiteral(currentTokenString.Value);
407  literal.Type = EnumTokenType.STRING_LITERAL;

414  tokens.AddLast(new LinkedListNode<Token>(literal));
