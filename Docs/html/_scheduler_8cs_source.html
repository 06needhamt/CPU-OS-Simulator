144  private async void RunScheduler()
145  {
146  while (readyQueue.Count > 0 && runningProcess == null)
147  {
148  switch (schedulingPolicy)
149  {
150  case EnumSchedulingPolicies.FIRST_COME_FIRST_SERVED:
151  runningProcess = readyQueue.Dequeue();
152  //ProcessExecutionUnit unit = runningProcess.Unit;
153  while (!runningProcess.Unit.Stop && !runningProcess.Unit.Done && !runningProcess.Unit.Process.Terminated)
154  {
155  runningProcess.Unit.ExecuteInstruction();
156  await CallFromMainThread(UpdateInterface);
157  }
158  if (runningProcess.Unit.Stop)
159  {
160  runningProcess.ProcessState = EnumProcessState.WAITING;
161  waitingQueue.Enqueue(runningProcess);
162  PCBFlags? flags = CreatePCBFlags(ref runningProcess);
163  if (flags != null)
164  {
165  runningProcess.ControlBlock = new ProcessControlBlock(flags.Value);
166  }
167  else
168  {
169  MessageBox.Show("There was an error while creating process control block flags");
170  return;
171  }
172  }
173  else
174  {
175  runningProcess = null;
176  }
177  break;
178  }
179  }
180  MessageBox.Show("OS Execution Complete");
181  return;
