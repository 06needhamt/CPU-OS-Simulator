129  private SchedulerFlags? CreateSchedulerFlags()
130  {
131  SchedulerFlags temp = new SchedulerFlags();
132  temp.schedulingPolicies = schedulingPolicy;
133  if (temp.schedulingPolicies == EnumSchedulingPolicies.ROUND_ROBIN) // if the user selected round robin scheduling
134  {
135  temp.RR_Priority_Policy = priorityPolicy;
136  temp.RR_TimeSlice = RR_Time_Slice;
137  temp.RR_Type = roundRobinType;
138  temp.TimeSliceUnit = RR_Time_Slice_Unit;
139  }
140  temp.allowCPUAffinity = allowCPUAffinity;
141  temp.defaultScheduler = useDefaultScheduler;
142  temp.runningWithNoProcesses = runWithNoProcesses;
143  if (!temp.runningWithNoProcesses) // if the user selected to run the scheduler no processes
144  {
145  temp.readyQueue = readyQueue;
146  temp.waitingQueue = waitingQueue;
147  SimulatorProcess proc = readyQueue.Dequeue();
148  if (proc != null)
149  {
150  proc.ProcessState = EnumProcessState.RUNNING;
151  }
152  temp.runningProcess = proc; // populate the queues and set the first processes state to running
153  }
154  else // otherwise create a blank queue and set the running process to null
155  {
156  readyQueue = new Queue<SimulatorProcess>();
157  waitingQueue = new Queue<SimulatorProcess>();
158  temp.runningProcess = null;
159  }
160  temp.cpuClockSpeed = CPUClockSpeed;
161  return temp;
162  }
