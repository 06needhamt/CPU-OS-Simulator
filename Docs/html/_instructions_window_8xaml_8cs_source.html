56  private void PopulateInstructions()
57  {
58  int count = Enum.GetNames(typeof(EnumOpcodes)).Length;
59  for (int i = 0; i < count; i++)
60  {
61  EnumOpcodes value = (EnumOpcodes)i;
62  string fulldescription = Extentions.DescriptionAttr<EnumOpcodes>(value);
63  string opcode = value.ToString();
64  string[] split = fulldescription.Split(new char[] { ':' }); // split the catagory from the description
65  string category = split[0];
66  string description = split[1];
67  instructionDescriptions.Add(description); // add the description to the list
68  switch (category)
69  {
70  //TODO Add other instruction Types
71  case "Data Transfer":
72  {
73  lst_OpcodeListDataTransfer.Items.Add(opcode); // populate data transfer instructions
74  break;
75  }
76  case "Logical":
77  {
78  lst_OpcodeListLogical.Items.Add(opcode); // populate logical instructions
79  break;
80  }
81  case "Arithmetic":
82  {
83  lst_OpcodeListArithmetic.Items.Add(opcode); //populate aritmetic instructions
84  break;
85  }
86  case "Control Transfer":
87  {
88  lst_OpcodeListControlTransfer.Items.Add(opcode); // populate control transfer instructions
89  break;
90  }
91  case "Comparison":
92  {
93  lst_OpcodeListComparison.Items.Add(opcode); // populate comparison instructions
94  break;
